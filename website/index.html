Get Started in Minutes

Choose your preferred installation method and start running a Neo-RS node today.

Download Latest Release 
# Download for Linux x86_64
curl -L https://github.com/r3e-network/neo-rs/releases/latest/download/neo-rs-linux-x86_64.tar.gz | tar xz
sudo mv neo-rs /usr/local/bin/

# Download for macOS
curl -L https://github.com/r3e-network/neo-rs/releases/latest/download/neo-rs-macos-x86_64.tar.gz | tar xz
sudo mv neo-rs /usr/local/bin/

# Download for Windows
# Download neo-rs-windows-x86_64.zip from GitHub releases

# Run TestNet node
neo-rs --testnet

# Run MainNet node  
neo-rs --mainnet
Build from Source 
# Install Rust (if not already installed)
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh
source $HOME/.cargo/env

# Install system dependencies
# Ubuntu/Debian:
sudo apt-get install librocksdb-dev pkg-config libssl-dev

# macOS:
brew install rocksdb

# Clone and build
git clone https://github.com/r3e-network/neo-rs.git
cd neo-rs
cargo build --release

# Run the node
./target/release/neo-rs --testnet
Docker Deployment 
# Quick start with Docker
docker run -d \
  --name neo-rs \
  -p 10332:10332 \
  -p 10333:10333 \
  -v neo_data:/data \
  neo/neo-rs:latest --testnet

# With custom configuration
docker run -d \
  --name neo-rs \
  -p 10332:10332 \
  -p 10333:10333 \
  -v ./neo-config.toml:/etc/neo/neo-config.toml:ro \
  -v neo_data:/data \
  neo/neo-rs:latest

# Docker Compose (full stack with monitoring)
curl -O https://raw.githubusercontent.com/neo-project/neo-rs/main/docker-compose.yml
docker-compose up -d
Kubernetes Deployment 
# Download Kubernetes manifests
curl -O https://raw.githubusercontent.com/neo-project/neo-rs/main/k8s/neo-rs-deployment.yaml
curl -O https://raw.githubusercontent.com/neo-project/neo-rs/main/k8s/neo-rs-service.yaml
curl -O https://raw.githubusercontent.com/neo-project/neo-rs/main/k8s/neo-rs-configmap.yaml

# Deploy to Kubernetes
kubectl apply -f neo-rs-configmap.yaml
kubectl apply -f neo-rs-deployment.yaml
kubectl apply -f neo-rs-service.yaml

# Check deployment status
kubectl get pods -l app=neo-rs
kubectl logs -l app=neo-rs -f

# Access RPC endpoint
kubectl port-forward service/neo-rs 10332:10332
